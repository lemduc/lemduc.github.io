function toggleTheme() {
  const newTheme = currentTheme === 'light' ? 'dark' : 'light';
  applyTheme(newTheme);
}

// CSS Custom Properties
:root {
  --bg-primary: #ffffff;
  --text-primary: #333333;

  --text-primary: #e0e0e0;
